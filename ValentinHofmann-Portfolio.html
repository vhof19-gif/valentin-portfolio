    const nextBtn = document.getElementById("nextBtn");

    const backToGrid = document.getElementById("backToGrid");


    let currentIndex = 0;


    function createThumb(i){

      const p = projects[i];

      const el = document.createElement("article");

      el.className = "thumb";

      el.tabIndex = 0;

      el.setAttribute("role","button");

      el.setAttribute("aria-label", `Open project: ${p.title}`);

      el.innerHTML = `

        <div class="thumbMedia" aria-hidden="true"></div>

        <div class="thumbMeta">

          <h3 class="thumbTitle">${p.title}</h3>

        </div>

      `;

      el.addEventListener("click", () => openProject(i, true));

      el.addEventListener("keydown", (e) => {

        if(e.key === "Enter" || e.key === " ") { e.preventDefault(); openProject(i, true); }

      });

      return el;

    }


    function renderGrid(){

      grid.innerHTML = "";

      projects.forEach((_, i) => grid.appendChild(createThumb(i)));

    }


    function makePlaceholderBlock(kind){

      const wrap = document.createElement("div");

      wrap.className = "ph" + (kind === "square" ? " square" : kind === "tall" ? " tall" : "");

      wrap.innerHTML = `<div class="phMedia" aria-hidden="true"></div>`;

      return wrap;

    }


    function openProject(i, scrollToTop){

      currentIndex = i;

      const p = projects[i];


      document.body.classList.add("projectActive");

      projectName.textContent = p.title;


      projectImages.innerHTML = "";

      p.layout.forEach(kind => projectImages.appendChild(makePlaceholderBlock(kind)));


      prevBtn.disabled = (i === 0);

      nextBtn.disabled = (i === projects.length - 1);


      if(scrollToTop){

        projectView.scrollIntoView({behavior:"smooth", block:"start"});

      }

      // Update URL hash for shareability (optional)

      history.replaceState(null, "", `#project-${i+1}`);

    }


    function closeProject(){

      document.body.classList.remove("projectActive");

      history.replaceState(null, "", "#portfolio");

      document.getElementById("portfolio").scrollIntoView({behavior:"smooth", block:"start"});

    }


    prevBtn.addEventListener("click", () => { if(currentIndex > 0) openProject(currentIndex - 1, true); });

    nextBtn.addEventListener("click", () => { if(currentIndex < projects.length - 1) openProject(currentIndex + 1, true); });

    backToGrid.addEventListener("click", (e) => { e.preventDefault(); closeProject(); });
